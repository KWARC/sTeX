\documentclass[mh]{smglom}
\libinput{preamble}
\begin{document}
\begin{modsig}[creators=miko,align=Sequences]{sequences}
\gimport[smglom/sets]{countable}
\gimport[smglom/sets]{finite-cardinality}
\MSC{11Bxx}

\symdef[noverb,noalign,bvars=2,bargs=3]{bseqon}[3]{{#2}_{#1\in #3}}
\symtest{bseqon}{\bseqon{i}{a_i}N}

\symdef[noverb,noalign,bvars=2,bargs=3]{sequenceon}[3]{{({#1}_{#2})}_{#2\in #3}}
\symtest[name=sequence]{sequenceon}{\sequenceon{a}iN}

\symdef[bvars=2,align]{sequence}[2]{({#1}_{#2})}
\symtest{sequence}{\sequence{a}i}

\symdef[noverb,noalign,assocarg=1]{seq}[1]{\assoc[nobrackets],{#1}}
\symtest{seq}{\seq{1,2,3,4}}

\symdef[noverb,noalign,assocarg=1]{seqfromto}[2]{\mixfixai[nobrackets]{}{#1}{,\ldots,}{#2}{},}
\symtest{seqfromto}{\seqfromto{1,2,3}{10}}

\symdef[noverb,noalign]{nsequi}[3]{\mixfixii[nobrackets]{}{{#1}^{#2}}{,\ldots,}{{#1}^{#3}}{}}
\symtest{nsequi}{\nsequi{x}1n}

\symdef[noverb,noalign]{nseqli}[3]{\mixfixii[nobrackets]{}{{#1}_{#2}}{,\ldots,}{{#1}_{#3}}{}}
\symtest{nseqli}{\nseqli{x}1n}

\symdef[name=infinite-sequence,assocarg=1]{infseq}[1]{\mixfixa[nobrackets]{}{#1}{,\ldots},}
\symtest[name=infinite-sequence]{infseq}{\infseq{1,2,3}}

\symdef[noverb,noalign]{infsequi}[3]{\mixfixi[nobrackets]{}{{#1}^{#2},{#1}^{#3}}{,\ldots}}
\symtest{infsequi}{\infsequi{x}12}

\symdef[noverb,noalign]{infseqli}[3]{\mixfixi[nobrackets]{}{{#1}_{#2},{#1}_{#3}}{,\ldots}}
\symtest{infseqli}{\infseqli{x}12}

\symdef[noverb,noalign]{seqsel}[2]{{#1}_{\withprecii\niprec #2}}
\symtest{seqsel}{\seqsel{S}i}

\symdef[noverb,noalign,assocarg=2]{seqExtend}[2]{\mixfixia[nobrackets]{}{#1},{#2}{},}
\symtest{seqExtend}{\seqExtend{S}3}

\symdef[name=length]{seqlenOp}{\text{len}}
\symdef[name=length]{seqlen}[1]{\prefix\seqlenOp{#1}}
\symtest{seqlen}{\seqlen\sigma}

\symii{finite}{sequence}
\end{modsig}

\end{document}
