\symdef[noverb,bvars=1,bargs={2,3},specializes=sum]{Sumfromto}[4]{\mixfixi[p=500]{}{\sum_{#1=#2}^{#3}{#4}}{}}
\symtest{Sumfromto}{\Sumfromto{i}1\infty{i}}

\symdef[noverb,specializes=sum,bvars=1,bargs=2]{SumColl}[3]{\sum_{#2}{\withpreciii{500}{#3}}}
\symtest{SumColl}{\SumColl{i}{i>0}{i}}

\symdef[noverb,specializes=sum,bvars=1]{SumOver}[2]{\sum_{#1}{\withprecii{500}{#2}}}
\symtest{SumOver}{\SumOver{i}i}

\symdef[noverb,specializes=sum,bvars=1,bargs=2]{SumInColl}[3]{\sum_{#1\in{#2}}{\withpreciii{500}{#3}}}
\symtest{SumInColl}{\SumInColl{i}{\mathbb{N}}{i}}

\symdef[noverb,specializes=sum,bvars=1,bargs=2]{SumProp}[3]{\sum_{#2}{\withprecii{500}{#3}}}
\symtest{SumProp}{\SumProp{i}{\text{$i\in\mathbb{N}$ prime}}{i^2}}
