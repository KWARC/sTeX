\documentclass[mh]{smglom}
\libinput{preamble}
\begin{document}
\begin{modsig}[creators=miko]{arithmetics}
\gimport[smglom/mv]{piecewise}
\gimport{numbers-orders}

\symdef[name=addition]{aplusOp}{\mathop{+}}
\symdef[assocarg=1,name=addition]{aplus}[1]{\assoc[p=500]{\aplusOp}{#1}}
\symtest{aplus}{\aplus{a,b,3}}

\symdef[name=subtraction]{aminusOp}{\mathop{-}}
\symdef[assocarg=1,name=subtraction]{aminus}[1]{\assoc[p=550]{\aminusOp}{#1}}
\symtest{aminus}{\aminus{a,b,3}}

\symdef[noverb,name=uminus]{auminusOp}{\mathop{-}}
\symdef[noverb,name=uminus]{auminus}[1]{\prefix[p=550]{\auminusOp}{#1}}
\symtest[name=uminus]{auminus}{\aminus{a}}

\symdef[name=absolute-value]{absolutevalue}[1]{\left|#1\right|}
\symtest{absolutevalue}{\absolutevalue{r}}

\symdef[name=multiplication]{atimesOp}{\mathop{\*}}
\symdef[assocarg=1,name=multiplication]{atimes}[1]{\assoc[p=600]{\atimesOp}{#1}}
\symtest{atimes}{\atimes{a,b,3}}

\symvariant{atimesOp}{cdot}{\mathop\cdot}
\symvariant{atimes}[1]{cdot}{\assoc[p=600]{\cdot}{#1}}
\symtest[variant=cdot]{atimes}{\atimes[cdot]{a,b,3}}

\symvariant{atimesOp}{x}{\mathop\times}
\symvariant{atimes}[1]{x}{\assoc[p=600]{\times}{#1}}
\symtest[variant=x]{atimes}{\atimes[x]{a,b,3}}

\symdef[name=division]{adivideOp}{\mathop{/}}
\symdef[name=division]{adivide}[2]{\infix[p=600]{\adivideOp}{#1}{#2}}
\symtest{adivide}{\adivide{n}2}

\symvariant{adivide}[2]{frac}{\mixfixi[nobrackets]{}{\frac{#1}{#2}}{}}
\symtest[variant=frac]{adivide}{\adivide[frac]{n}2}

\symdef[name=exponentiation]{power}[2]{\mixfixi[p=800,pi=700]{}{#1^{\withprecii\niprec #2}}{}}
\symvariant{power}[2]{basebrack}{\mixfixi[p=800,pi=700]{}{(#1)^{\withprecii\niprec #2}}{}}
\symtest{power}{\power{3}{4}=81}
\symtest[variant=basebrack]{power}{\power[basebrack]{-3}{4}=81}

\symdef[name=root]{aroot}[2]{\sqrt[#1]{#2}}
\symtest{aroot}{\aroot37}

\symdef[name=square-root]{asqrt}[1]{\sqrt{#1}}
\symtest{asqrt}{\asqrt{7}}
\symi[gfc=N]{exponent}
\symi[gfc=N]{power}
\symi[gfc=N]{base}
\symi[gfc=N]{sum}
\symi[gfc=N]{difference}
\symi[gfc=N]{product}
\symi[gfc=N]{quotient}
\end{modsig}
\end{document}
