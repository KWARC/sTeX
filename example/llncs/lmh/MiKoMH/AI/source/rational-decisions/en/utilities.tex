  \begin{itemize}
  \item \Trefi[decision-theory?utility-function]{utilities} map states to real numbers. Which
    numbers?
  \item
    \begin{definition}[title=Standard approach to assessment of human utilities,id=standardlottery.def]
      Compare a given state $A$ to a \defii{standard}{lottery} $\standardlottery{p}$ that
      has
      \begin{itemize}
      \item ``best possible prize'' $\ubest$ with probability $p$
      \item ``worst possible catastrophe'' $\uworst$ with probability $1-p$
      \end{itemize}
      adjust \trefi[preferences]{lottery} probability $p$ until
      $\ratindiff{A}{\standardlottery{p}}$.  Then $\utilityof{A}=p$.
    \end{definition}
  \item 
    \begin{example}[for=standardlottery.def]
      Choose $\ubest$ \hateq current state, $\uworst$ \hateq instant death

    \end{example}
  \end{itemize}
