  \frametitle{Agents and Environments}
  \begin{itemize}
  \item
    \begin{definition}
      An \defi{agent} is anything that
      \begin{itemize}
      \item perceives its \defi{environment} via \defis{sensor} (means of sensing the
        \trefi{environment})
      \item acts on it with \defis{actuator} (means of changing the environment).
        \cmhgraphics[width=.65\slidewidth]{rational-agents/PIC/agent-environment} 
      \end{itemize}
    \end{definition}
  \item
    \begin{example}
      \Trefis{agent} include humans, robots, softbots, thermostats, etc.
    \end{example}
  \end{itemize}
