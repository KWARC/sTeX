\begin{frame}
  \frametitle{The Math of Problem Formulation: Search Problems}
  \begin{itemize}
  \item
    \begin{definition}[id=search-problem.def]
      A \defii{search}{problem} $\defeq\Pi{\mvstructure{\cS,\cA,\cT,\cI,\cG}}$ consists of
      a set $\cS$ of \defis{state}, a set $\cA$ of \defis{action}, and a
      \defii{transition}{model} $\fun\cT{\pairs\cA\cS}{\powerset\cS}$ that assigns to any
      action $\inset{a}\cA$ and state $\inset{s}\cS$ a set of \defiis{successor}{state}.

      Certain states in $\cS$ are designated as \defiis{goal}{state} ($\sseteq\cG\cS$) and
      \defiis{initial}{state} $\sseteq\cI\cS$.
    \end{definition}
  \item 
    \begin{definition}
      We say that an \trefi{action} $\inset{a}\cA$ is \defi{applicable} in a \trefi{state}
      $\inset{s}\cS$, iff $\nappa\cT{a,s}\ne\eset$. We call
      $\fun{\SPresultRelation\cT{a}}\cS{\powerset\cS}$ with
      $\defeq{\nappa{\SPresultRelation\cT{a}}{s}}{\nappa\cT{a,s}}$ the
      \defii{result}{relation} for $a$ and
      $\defeq{\SPresultRelation\cT\cA}{\munionCollection{a}\cA{\SPresultRelation\cT{a}}}$ the
      \trefii{result}{relation} of $\Pi$.
    \end{definition}
  \item
    \begin{definition}[id=search-solution.def]
      A \defi{solution} for a \trefii{search}{problem} $\mvstructure{\cS,\cA,\cT,\cI,\cG}$
      consists of a \trefi[sequences]{sequence} of \trefis{action} $\nseqli{a}1n$ such
      that for all $\natbetweenss1in$
      \begin{itemize}
      \item $\livar{a}i$ is \trefi{applicable} to \trefi{state}
        $\livar{s}{i-1}$, where $\inset{\livar{s}0}\cI$,
      \item $\inset{\livar{s}i}{\nappa{\SPresultRelation\cT{\livar{a}i}}{\livar{s}{i-1}}}$,
        and $\inset{\livar{s}{n}}\cG$.
      \end{itemize}
    \end{definition}
  \item
    \begin{omtext}[title=Idea]
      A \trefi{solution} bring us from $\cI$ to a \trefii{goal}{state}.
    \end{omtext}
  \item
    \begin{definition}
      Often we add a \defii{cost}{function} $\fun\opcostFN\cA\NonNegativeRealNumbers$ that
      associates a \defii{step}{cost} $\opcost{a}$ to an \trefi{action}
      $\inset{a}\cA$. The \defi{cost} of a \trefi{solution} is the sum of the
      \trefiis{step}{cost} of its \trefis{action}.
    \end{definition}
  \end{itemize}
\end{frame}

\begin{nomtext}[title=Observation]
  The \trefi[problem-formulation?problem-formulation]{problem} of problems from
  \sref{search-problem.def} is essentially a ``black-box'' description that provides the
  functionality needed to construct the state space. We will come back to this in slide
  \ref{slide.blackwhitebox}. 
\end{nomtext}
