\documentclass{article}
\usepackage{cpath}

\def\result{%
\ifpath%
Equal.%
\else%
Not Equal.%
\fi%
}%

\begin{document}
\begin{enumerate}
	\item \ComparePath{/foo/bar/../baz}{/../../bar/../foo/baz}\result % both are /foo/baz
		\newline\cpath{/foo/bar/../baz}
		\newline\cpath{/../../bar/../foo/baz}
	\item \ComparePath{/test}{test/a}\result
	\item \ComparePath{/../foo/../..}{/}\result % two root dir
	\item \ComparePath{/a/b/../c}{/a/c/../b}\result
	\item \ComparePath{foo/bar/baz}{/foo/bar/baz}\result %treat equal
	\item \ComparePath{/foo///baz}{../foo/baz}\result % also support some non-standard way
\end{enumerate}
\end{document}
