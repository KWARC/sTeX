%% Define the key by \define@key
%% Use the key by \setkeys
\documentclass{article}
\usepackage{keyval}
\makeatletter
\def\try[#1]#2{{\setkeys{group}{#1}input: #2}}
\define@key{group}{scale}{scale: \texttt{#1}\newline}
\define@key{group}{clip}[true]{clip: \texttt{#1}\newline}
\makeatother
\begin{document}
\noindent
\try[scale=3in,clip=false]{aaa}
\end{document}
